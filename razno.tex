    \begin{itemize}
    \item[\rm (i)] $N$ je edinka.
    \item[\rm (ii)] $aN \subseteq Na$ za vsak $a \in G$.

    \item[\rm (iv)] $aNa^{-1} = N$ za vsak $a \in G$. 
    \end{itemize}
\end{trditev}
